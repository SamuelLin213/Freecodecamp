<!---in overlapping elements(positioned using position property) the later appearing element will by default appear on top, z-index property can specify order of how elements are stacked(higher integer  moves it higher on stack than lower values)--->

<style>
  div {
    width: 60%;
    height: 200px;
    margin-top: 20px;
  }

  .first {
    background-color: red;
    position: absolute;
    z-index: 2;
  }
  .second {
    background-color: blue;
    position: absolute;
    left: 40px;
    top: 50px;
    z-index: 1;
  }
</style>

<div class="first"></div>
<div class="second"></div>
